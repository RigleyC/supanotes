Append to `backend/db/queries/ai.sql`:

```sql
-- name: GetRetryableEmbeddings :many
SELECT n.id, n.content, n.user_id
FROM notes n
WHERE n.deleted_at IS NULL
  AND NOT n.is_inbox
  AND (
    n.embedding_status = 'pending'
    OR (n.embedding_status = 'failed' AND n.updated_at < NOW() - INTERVAL '5 minutes')
  )
ORDER BY n.embedding_status = 'pending' DESC, n.updated_at ASC
LIMIT $1;
```

const expectedEmbeddingDim = 1536

func (c *EmbeddingClient) GenerateEmbedding(ctx context.Context, text string) ([]float64, error) {
    if c.apiKey == "" {
        return makeMockEmbedding(), nil
    }

    var lastErr error
    for attempt := 0; attempt < 3; attempt++ {
        if attempt > 0 {
            time.Sleep(time.Duration(1<<attempt) * time.Second)
        }
        emb, err := c.generateEmbeddingOnce(ctx, text)
        if err == nil {
            return emb, nil
        }
        if !isRetryableError(err) {
            return nil, err
        }
        lastErr = err
    }
    return nil, fmt.Errorf("embeddings: max retries exceeded: %w", lastErr)
}

func (c *EmbeddingClient) generateEmbeddingOnce(ctx context.Context, text string) ([]float64, error) {

func isRetryableError(err error) bool {
    if err == nil {
        return false
    }
    // Simple heuristic: check if error message contains HTTP status codes
    s := err.Error()
    return strings.Contains(s, "429") || strings.Contains(s, "500") || strings.Contains(s, "503")
