**Goal:** Execute the critical follow-up tasks for the Node-Tree Architecture shift: adapting the RAG embedding pipeline, creating a robust data migration script for production users, deleting legacy markdown serialization code from the frontend, removing the redundant `SaveThrottle`, and simplifying the agent's write tools to prevent backend markdown-parsing overengineering.

**Architecture:** 
- RAG embeddings are generated by concatenating the node tree into a flat string for the embedding model.
- The migration script parses existing Markdown to populate `note_nodes`.
- The Agent's tools are simplified: `update_note` is deprecated. Instead, the agent writes using specific tools: `append_to_note` (inserts a paragraph node), `create_task`, and `update_task`.
- The frontend editor's `SaveThrottle` is removed, saving changes synchronously.

**Tech Stack:** Go (Backend), Flutter (Frontend)

```go
package notes

import (
	"context"
	"strings"

	"github.com/pgvector/pgvector-go"
	"github.com/RigleyC/supanotes/internal/db/sqlcgen"
	"github.com/RigleyC/supanotes/pkg/llm"
)

type EmbeddingService struct {
	q       sqlcgen.Querier
	embedCL *llm.EmbeddingClient
}

func NewEmbeddingService(q sqlcgen.Querier, embedCL *llm.EmbeddingClient) *EmbeddingService {
	return &EmbeddingService{q: q, embedCL: embedCL}
}

func (s *EmbeddingService) RegenerateNoteEmbedding(ctx context.Context, noteID string) error {
	nodes, err := s.q.GetNodesByNoteId(ctx, noteID)
	if err != nil {
		return err
	}

	var sb strings.Builder
	for _, n := range nodes {
		sb.WriteString(string(n.Data) + "\n") 
	}
	content := sb.String()

	emb, err := s.embedCL.GenerateEmbedding(ctx, content)
	if err != nil {
		return err
	}

	vec := make([]float32, len(emb))
	for i := range emb {
		vec[i] = float32(emb[i])
	}
	
	return s.q.UpdateNoteSearchVector(ctx, sqlcgen.UpdateNoteSearchVectorParams{
		ID:           noteID,
		SearchVector: pgvector.NewVector(vec),
	})
}
```

### Task 2: Data Migration Script

**Files:**
- Create: `backend/cmd/migrate_nodes/main.go`

- [ ] **Step 1: Write Migration CLI**

Create `backend/cmd/migrate_nodes/main.go`:
```go
package main

import (
	"context"
	"log"
	"os"
	"strings"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5/pgxpool"
	"github.com/RigleyC/supanotes/internal/db/sqlcgen"
)

func main() {
	ctx := context.Background()
	dbURL := os.Getenv("DATABASE_URL")
	pool, err := pgxpool.New(ctx, dbURL)
	if err != nil {
		log.Fatal(err)
	}
	q := sqlcgen.New(pool)

	notes, err := q.GetAllNotesForMigration(ctx)
	if err != nil {
		log.Fatal(err)
	}

	for _, note := range notes {
		content := note.Content
		if content == "" {
			continue
		}
		
		lines := strings.Split(content, "\n")
		for i, line := range lines {
			nodeType := "paragraph"
			if strings.HasPrefix(strings.TrimSpace(line), "- [") {
				nodeType = "task"
			}
			
			_, err := q.InsertNode(ctx, sqlcgen.InsertNodeParams{
				ID:       uuid.NewString(),
				NoteID:   note.ID,
				Position: int32(i),
				Type:     nodeType,
				Data:     []byte(`{"text":"` + line + `"}`),
			})
			if err != nil {
				log.Printf("Error inserting node for note %s: %v", note.ID, err)
			}
		}
	}
	log.Println("Migration complete")
}
```
