  // Called when a push arrives while the app is in background/killed

  await Firebase.initializeApp();

- [ ] **Step 2: Call it after login/register in `auth_controller.dart`**

```dart
import 'package:firebase_messaging/firebase_messaging.dart';

Future<AuthResult> login({...}) async {
  state = const AsyncValue.loading();
  try {
    final result = await _repository.login(email: email, password: password);
    await _sessionCache.hydrate({...});
    state = AsyncValue.data(result.user);

    // Register FCM token (non-blocking)
    try {
      final fcmToken = await FirebaseMessaging.instance.getToken();
      if (fcmToken != null) {
        await _repository.registerDeviceToken(fcmToken);
      }
    } catch (_) {
      // Non-fatal — push not available in dev/simulator
    }

    return result;
  } catch (e, st) {
    state = AsyncValue.error(e, st);
    rethrow;
  }
}
```

- [ ] **Step 1: Create `NoteTagsChipBar` widget**

`lib/features/notes/presentation/widgets/note_tags_chip_bar.dart`:

```dart
library;

import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:supanotes/core/database/daos/tags_dao.dart';

final tagsForNoteProvider = StreamProvider.family<List<TagData>, String>(
  (ref, noteId) => ref.watch(tagsDaoProvider).watchTagsForNote(noteId),
);

class NoteTagsChipBar extends ConsumerWidget {
  final String noteId;

  const NoteTagsChipBar({super.key, required this.noteId});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final tagsAsync = ref.watch(tagsForNoteProvider(noteId));

    return tagsAsync.when(
      data: (tags) {
        if (tags.isEmpty) return const SizedBox.shrink();
        return Padding(
          padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 4),
          child: Wrap(
            spacing: 6,
            runSpacing: 4,
            children: tags.map((tag) {
              return Chip(
                label: Text(tag.name, style: const TextStyle(fontSize: 12)),
                deleteIcon: const Icon(Icons.close, size: 16),
                onDeleted: () {
                  ref.read(tagsDaoProvider).detachTag(noteId, tag.id);
                },
                materialTapTargetSize: MaterialTapTargetSize.shrinkWrap,
              );
            }).toList(),
          ),
        );
      },
      loading: () => const SizedBox.shrink(),
      error: (_, __) => const SizedBox.shrink(),
    );
  }
}

// Add tagsDaoProvider to core/di/providers.dart if not exists
final tagsDaoProvider = Provider((ref) => ref.watch(appDatabaseProvider).tagsDao);
```

Future<void> _showTagPicker(BuildContext context) async {
  final tags = await context.read(tagsDaoProvider).getAllTags();
  if (!context.mounted) return;
  final selected = await showDialog<String>(
    context: context,
    builder: (ctx) => SimpleDialog(
      title: const Text('Adicionar tag'),
      children: tags
          .map((t) => SimpleDialogOption(
                onPressed: () => Navigator.pop(ctx, t.id),
                child: Text(t.name),
              ))
          .toList(),
    ),
  );
  if (selected != null) {
    context.read(tagsDaoProvider).attachTag(widget.noteId, selected);
  }
}
